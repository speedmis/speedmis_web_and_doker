--select sql :
select zDBseobeo from (select distinct 2 as nnn, case when ifnull(table_m.dbalias,'')='' then '기본MYSQL' else table_m.dbalias end as "zDBseobeo" from MisMenuList table_m 
left outer join MisMenuList table_g2 on table_g2.autogubun = left(table_m.autogubun,2)
left outer join MisCommonTable table_MenuType on table_MenuType.kcode = table_m.MenuType
 and table_MenuType.gcode='speedmis000306'
left outer join MisUser table_wdater on table_wdater.UniqueNum = table_m.wdater
left outer join MisUser table_lastupdater on table_lastupdater.UniqueNum = table_m.lastupdater
left outer join MisUser table_filelastupdater on table_filelastupdater.UniqueNum = table_m.filelastupdater
where 9=9  and table_m.useflag='1'
and table_m.MenuType = '01') aaa order by nnn, zDBseobeo limit 31

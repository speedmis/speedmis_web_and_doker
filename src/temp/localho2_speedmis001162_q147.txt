select ic3 from (select distinct 2 as nnn, table_m.ic3 as "ic3" from edu_N_ghsa_new table_m 
left outer join MisUser table_uniqueNum on table_uniqueNum.uniqueNum=table_m.gh_id
left outer join MisCommonTable table_gh3 on table_gh3.kcode = table_m.gh3
 and (table_gh3.gcode='edu001062')
left outer join MisUser table_gh_id on table_gh_id.uniquenum=table_m.gh_id
left outer join MisUser table_wdater on table_wdater.UniqueNum = table_m.wdater
left outer join MisUser table_lastupdater on table_lastupdater.UniqueNum = table_m.lastupdater
where 9=9  and table_m.useflag='1'
) aaa order by nnn, ic3 limit 31

select gname from (select distinct 1 as nnn, table_m.gname as "gname" from MisGroup_Master table_m 
left outer join MisUser table_wdater on table_wdater.UniqueNum = table_m.wdater
left outer join MisUser table_lastupdater on table_lastupdater.UniqueNum = table_m.lastupdater
where 9=9  and table_m.useflag='1'
 and table_m.gname like N'%운영자1%' union select distinct 2 as AAA, table_m.gname as "gname" from MisGroup_Master table_m  
left outer join MisUser table_wdater on table_wdater.UniqueNum = table_m.wdater
left outer join MisUser table_lastupdater on table_lastupdater.UniqueNum = table_m.lastupdater
where 9=9  and table_m.useflag='1'
 and table_m.gname not like N'%운영자1%') aaa order by nnn, gname limit 31

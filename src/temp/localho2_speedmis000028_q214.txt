select table_upRealPidQnMenuName from (select distinct 2 as nnn, table_upRealPid.MenuName as "table_upRealPidQnMenuName" from MisBoardArticle table_m 
left outer join MisMenuList_Member table_member on table_member.RealPid = table_m.RealPid and table_member.userid = 'gadmin'
left outer join MisMenuList table_RealPid on table_RealPid.RealPid = table_m.RealPid and MenuType in ('01','06') and (ifnull(table_RealPid.AuthCode,'')<>'02' or table_member.userid = 'gadmin')
left outer join MisMenuList table_upRealPid on table_upRealPid.RealPid = table_RealPid.upRealPid
left outer join MisUser table_wdater on table_wdater.UniqueNum = table_m.wdater
left outer join MisUser table_lastupdater on table_lastupdater.UniqueNum = table_m.lastupdater
left outer join MisStation table_Station_NewNum on table_wdater.station_newnum=table_Station_NewNum.num
where 9=9  and table_m.useflag='1'
and table_RealPid.MenuType in ('01','06') and (ifnull(table_RealPid.AuthCode,'')<>'02' or table_member.userid = 'gadmin' and table_member.AuthorityLevel<>9) and table_RealPid.useflag='1' and table_upRealPid.MenuName<>'고객센터') aaa order by nnn, table_upRealPidQnMenuName limit 31
